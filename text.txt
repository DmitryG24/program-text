#include <iostream>
#include <time.h>
#define size 25
//Версия в ветке

int input_int()
{
	int number;
	int itsokay = 1;
	while (itsokay)
	{
		if (scanf_s("%d", &number) == 0 || number <= 3)
		{
			while (getchar() != '\n');
			puts("Ошибка - повторите ввод");
		}
		else
			itsokay = 0;
	}
	return number;
}

int main()
{
	setlocale(LC_ALL, "rus");
	int arr[size];
        printf("Поменять местами 4 и минимальный элементы массива");
	printf("Введите количество элементов массива: ");
	int n = input_int();
	
	printf("Исходный массив\n");
	srand(time(0));
	for (int i = 0; i < n; i++)
		arr[i] = rand() % 20;

	for (int i = 0; i < n; i++)
		printf("%d ", arr[i]);
	printf("\n");
	int min = arr[0];
	int minIndex = 0;
	for (int i = 0; i < n; i++)
	{
		if (min > arr[i])
		{
			min = arr[i];
			minIndex = i;
		}
	}

	double temp = arr[minIndex];
	arr[minIndex] = arr[3];
	arr[3] = temp;


        printf("Полученный массив\n");
	for (int i = 0; i < n; i++)
	{
		printf("%d ", arr[i]);
	}
	return 0;
}
